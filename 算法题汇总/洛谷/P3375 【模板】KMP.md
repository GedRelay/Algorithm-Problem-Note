# 思路
- 思路 1：
KMP 模板题


```cpp
#include<iostream>
using namespace std;

string s, p; // 主串和模式串
int nxt[1000010];

int main()
{
    cin >> s >> p;
    // 求next数组
    for(int i = 1, k = 0; i < p.size(); i++){
        while(k != 0 && p[i] != p[k]) k = nxt[k - 1];
        if(p[i] == p[k]) k++;
        nxt[i] = k;
    }
    // 进行匹配
    for(int i = 0, j = 0; i <s.size(); i++){ // i 指向主串, j 指向模式串
        while(j != 0 && s[i] != p[j]) j = nxt[j - 1];
        if(s[i] == p[j]) j++;
        if(j == p.size()){
            cout << i - j + 2 << endl;
            j = nxt[j - 1];
        }
    }
    // 输出next数组
    for(int i = 0; i < p.size(); i++){
        cout << nxt[i] << " ";
    }
	return 0;
}
```


- 时间复杂度：${O\left( n+m \right)  }$ 
- 空间复杂度：${O\left( m \right)  }$
