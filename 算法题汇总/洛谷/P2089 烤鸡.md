#include<cstdio>
#include<iostream>
#define fo(i,a,b) for(int i=a;i<=b;i++)
using namespace std;
int main()
{
	int n, ans = 0;
	cin >> n;
	if (n > 30)
	{
		cout << "0";
		return 0;
	}
	fo(a, 1, 3) fo(b, 1, 3) fo(c, 1, 3) fo(d, 1, 3) fo(e, 1, 3) fo(f, 1, 3)
		fo(g, 1, 3) fo(h, 1, 3) fo(i, 1, 3) fo(j, 1, 3)
		if (a + b + c + d + e + f + g + h + i + j == n)
			ans++;
	printf("%d\n", ans);
	fo(a, 1, 3) fo(b, 1, 3) fo(c, 1, 3) fo(d, 1, 3) fo(e, 1, 3) fo(f, 1, 3)
		fo(g, 1, 3) fo(h, 1, 3) fo(i, 1, 3) fo(j, 1, 3)
		if (a + b + c + d + e + f + g + h + i + j == n)
			printf("%d %d %d %d %d %d %d %d %d %d\n", a, b, c, d, e, f, g, h, i, j);
	return 0;
}

```cpp
#include<iostream>
#include<vector>
using namespace std;

int n;
vector<vector<int>> ans;
vector<int> path;

void dfs(int x, int sum){
    if(sum > n) return; // 剪枝
    if(x > 10){
        if(sum == n) ans.push_back(path);
        return;
    }
    for(int i = 1; i <= 3; i++){
        path.push_back(i);
        dfs(x + 1, sum + i);
        path.pop_back();
    }
}

int main()
{
	cin >> n;
	dfs(1, 0);
	cout << ans.size() << endl;
	for(int i = 0; i < ans.size(); i++){
	    for(int j = 0; j < 10; j++){
	        cout << ans[i][j] << " ";
	    }
	    cout << endl;
	}
	return 0;
}
```
