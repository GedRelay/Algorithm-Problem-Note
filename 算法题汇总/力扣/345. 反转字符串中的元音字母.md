# 分类
#双指针/两边收缩 

# 思路
- 思路 1：
两边收缩的双指针

```cpp
class Solution {
public:
    string reverseVowels(string s) {
        unordered_set<char> st{'a', 'e', 'i', 'o', 'u', 'A', 'E', 'I', 'O', 'U'};
        for(int l = 0, r = s.size() - 1; l < r; l++, r--){
            while(l < s.size() && !st.count(s[l])) l++;
            while(r >= 0 && !st.count(s[r])) r--;
            if(l < r) swap(s[l], s[r]);
        }
        return s;
    }
};
```


- 时间复杂度：${O\left( n \right)  }$ 
- 空间复杂度：${O\left( 1 \right)  }$
