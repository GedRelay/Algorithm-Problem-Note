int main()

	int n;

	for (int k = 1;; k++)
	{
		int m = n - 1092 * k;
		if (m % 364 == 0)
		{
			if (1 <= m / 364 && m / 364 <= 100)
			{
				cout << m / 364 << endl << k;
				return 0;
			}
			else continue;
		}
	}
}
