# 思路
- 思路 1：
结构体排序


```cpp
#include<iostream>
#include<algorithm>
using namespace std;
struct stu {
	int num;
	int s;
};
bool cmp(stu a, stu b) {
	if (a.s != b.s)return a.s > b.s;
	return a.num < b.num;
}
int main() {
	int n, m;
	cin >> n >> m;
	stu a[5010];
	int act = (m * 1.5) / 1;
	for (int i = 0; i < n; i++) {
		cin >> a[i].num >> a[i].s;
	}
	sort(a, a + n, cmp);
	int tmp = act;
	for (int i = act - 1; i < n; i++) {
		if (a[i].s == a[i + 1].s)tmp++;
	}
	cout << a[act - 1].s << " " << tmp << endl;
	for (int i = 0; i < tmp; i++) {
		cout << a[i].num << " " << a[i].s << endl;
	}
	return 0;
}
```


- 时间复杂度：${O\left( n\log n \right)  }$ 
- 空间复杂度：${O\left( n \right)  }$
