```cpp
#include<cmath>
#include<algorithm>
#include<iostream>
using namespace std;
int a[1023], b[1023];
int main()
{
	int n;
	cin >> n;
	for (int i = 0; i < n; i++)
		cin >> a[i];
	for (int i = 0; i < n - 1; i++)
		b[i] = abs(a[i] - a[i + 1]);
	sort(b, b + n - 1);
	for(int i=1;i<=n-1;i++)
		if (b[i - 1] != i)//不对
		{
			cout << "Not jolly";
			return 0;
		}
	cout << "Jolly";
	return 0;
}
```


- 时间复杂度：${O\left( n\log n \right)  }$ 
- 空间复杂度：${O\left( n \right)  }$
