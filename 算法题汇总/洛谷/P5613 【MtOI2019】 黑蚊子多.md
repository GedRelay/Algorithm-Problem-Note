# 思路
- 思路 1：


```cpp
#include<iostream>
using namespace std;
int main()
{
	int n, m, k,a[12],me=0,step=0,c=1;//n为总台阶数，m为步长，k为特殊阶梯个数,me记录所在位置
	cin >> n >> m >> k;
	for (int i = 1; i <= k; i++)
	{
		cin >> a[i];//ai存的是特殊阶梯的标号
	}
	while (me < n)
	{
		me += m;
		step++;
		if (k != 0)
		{
			while (a[c]<me)
			{
				c++;
			}
		}

		if (me == a[c])m++;
	}
	cout << step;
	return 0;
}
```
