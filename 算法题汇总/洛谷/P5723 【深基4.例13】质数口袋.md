# 思路
- 思路 1：


```cpp
#include<bits/stdc++.h>
using namespace std;

int L;
bool st[100010];

vector<int> get_primes(int n){
    vector<int> primes;
    for(int i = 2; i <= n; i++){
        if(!st[i]){
            primes.push_back(i);
        }
        for(int j = 0; primes[j] <= n / i; j++){
            st[primes[j] * i] = true;
            if(i % primes[j] == 0) break;
        }
    }
    return primes;
}

int main(){
    cin >> L;
    vector<int> primes = get_primes(100010);
    int sum = 0, cnt = 0;
    while(sum + primes[cnt] <= L){
        cout << primes[cnt] << endl;
        sum += primes[cnt];
        cnt++;
    }
    cout << cnt;
    return 0;
}
```


- 时间复杂度：${O\left( L \right)  }$ 
- 空间复杂度：${O\left( L \right)  }$
