# 思路
- 思路 1：
枚举所有分界点，并统计需要修改多少个格子


```cpp
#include<iostream>
using namespace std;

int n, m;
char mp[52][52];

int main()
{
	cin >> n >> m;
	for(int i = 1; i <= n; i++){
	    for(int j = 1; j <= m; j++){
	        cin >> mp[i][j];
	    }
	}
	int ans = 1e9;
	for(int i = 1; i <= n - 2; i++){
	    for(int j = i + 1; j <= n - 1; j++){
	        // 1~i为W, i+1~j为B, j+1~n为R
	        int cnt = 0; // 需要涂色的数量
	        for(int x = 1; x <= i; x++){
	            for(int y = 1; y <= m; y++){
	                if(mp[x][y] != 'W') cnt++;
	            }
	        }
	        for(int x = i + 1; x <= j; x++){
	            for(int y = 1; y <= m; y++){
	                if(mp[x][y] != 'B') cnt++;
	            }
	        }
	        for(int x = j + 1; x <= n; x++){
	            for(int y = 1; y <= m; y++){
	                if(mp[x][y] != 'R') cnt++;
	            }
	        }
	        ans = min(ans, cnt);
	    }
	}
	cout << ans;
	return 0;
}
```


- 时间复杂度：${O\left( n^{3} m \right)  }$ 
- 空间复杂度：${O\left( nm  \right)  }$
