# 思路
- 思路 1：
搜索所有可能的路线，找到一个路线答案 $+1$ 即可



```cpp
#include<bits/stdc++.h>
using namespace std;
int n, m, t;
int sx, sy, fx, fy;
bool mp[10][10];//1表示障碍物，0表示通路
bool vis[10][10];
int dx[4] = { -1,0,1,0 };
int dy[4] = { 0,1,0,-1 };
int ans;
void dfs(int px,int py)
{
	if (px == fx && py == fy)
		ans++;
	for (int i = 0; i < 4; i++)//搜索四个方向
	{
		int nx = px + dx[i], ny = py + dy[i];
		if (nx >= 1 && nx <= n && ny >= 1 && ny <= m && !vis[nx][ny] && !mp[nx][ny])
		{
			vis[nx][ny] = 1;
			px += dx[i], py += dy[i];
			dfs(px,py);
			vis[nx][ny] = 0;
			px -= dx[i], py -= dy[i];
		}
	}
}
int main()
{
	cin >> n >> m >> t;
	cin >> sx >> sy >> fx >> fy;
	int px = sx, py = sy;
	vis[sx][sy] = 1;
	while (t--){
		int aaa, bbb;
		cin >> aaa >> bbb;
		mp[aaa][bbb] = 1;
	}
	dfs(px, py);
	cout << ans;
	return 0;
}
```
