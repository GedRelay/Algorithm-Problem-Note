# 思路
- 思路 1：


```cpp
#include<bits/stdc++.h>
using namespace std;
int n;
int ans[20];
bool vis[20];
void dfs(int x)
{
	if (floor > n)
	{
		for (int i = 1; i <= n; i++)
			printf("%5d", ans[i]);
		printf("\n");
		return;
	}
	for (int i = 1; i <= n; i++)
	{
		if (!vis[i])
		{
			vis[i] = 1;
			ans[x] = i;
			dfs(x + 1);
			vis[i] = 0;
		}
	}

}
int main()
{
	cin >> n;
	dfs(1);
	return 0;
}
```
