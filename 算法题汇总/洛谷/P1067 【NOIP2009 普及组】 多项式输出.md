```cpp
#include<iostream>
using namespace std;
int main()
{
	int n;
	cin >> n;
	for (int i = n; i >=0; i--){
		int a;
		cin >> a;
		if (a == 0)
			continue;
		if (i != n && a > 0)
			cout << "+";
		if (a == -1)
			cout << "-";
		if (a != 1 && a != -1)
			cout << a;
		if (i != 0)
			cout << "x";
		if (i != 1 && i != 0)
			cout << "^" << i;
		if (i == 0 && (a == -1 || a == 1))
			cout << "1";
	}
	return 0;
}
```


- 时间复杂度：${O\left( n \right)  }$ 
- 空间复杂度：${O\left( n \right)  }$
