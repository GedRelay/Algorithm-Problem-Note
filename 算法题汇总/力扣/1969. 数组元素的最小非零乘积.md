# 思路
- 思路 1：


```cpp
class Solution {
public:
    const int MOD = 1e9+7;
    long long pow(long long a, long long b){
        long long res = 1, t = a % MOD;
        while(b){
            if(b & 1) res = res * t % MOD;
            t = t * t % MOD;
            b >>= 1;
        }
        return res % MOD;
    }

    int minNonZeroProduct(int p) {
        long long k = (1LL << p) - 1;
        long long y = 1LL << (p - 1);
        return k % MOD * pow(k - 1, y - 1) % MOD;
    }
};
```


- 时间复杂度：${O\left( p \right)  }$ 
- 空间复杂度：${O\left( p \right)  }$
