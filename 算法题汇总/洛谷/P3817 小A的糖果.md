# 思路
- 思路 1：


```cpp
#include<iostream>
using namespace std;

int n, k;
int a[100010];

int main()
{
	cin >> n >> k;
	for(int i = 1; i <= n; i++) cin >> a[i];
	long long ans = 0;
	for(int i = 1; i <= n; i++){
	    if(a[i] + a[i + 1] > k){
	        int t = a[i] + a[i + 1] - k; // 总共要减去的数量
	        if(a[i + 1] >= t) a[i + 1] -= t; // 后面的直接减去t个糖
	        else a[i + 1] = 0; // 后面的减为0，本身减去剩下的几颗糖
	        ans += t;
	    }
	}
	cout << ans;
	return 0;
}
```


- 时间复杂度：${O\left( n \right)  }$ 
- 空间复杂度：${O\left( n \right)  }$
