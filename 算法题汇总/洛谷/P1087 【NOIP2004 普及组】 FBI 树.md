# 思路
- 思路 1：


```cpp
#include<iostream>
using namespace std;
void f(string s)
{
	if (s.size() == 1)
		cout << (s[0] == '0' ? "B" : "I");
	else{
		string l = s.substr(0, s.size() / 2);
		string r = s.substr(s.size() / 2);
		f(l);
		f(r);
		bool B = 1, I = 1;
		for (int i = 0; s[i]; i++) {
			if (s[i] == '0')I = 0;
			else B = 0;
		}
		if (I)cout << "I";
		else if (B)cout << "B";
		else cout << "F";
	}
}
int main()
{
	int n;
	string s;
	cin >> n >> s;
	f(s);
	return 0;
}
```


- 时间复杂度：${O\left( n \right)  }$ 
- 空间复杂度：${O\left( n \right)  }$
