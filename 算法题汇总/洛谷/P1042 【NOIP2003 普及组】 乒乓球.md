# 思路
- 思路 1：


```cpp
#include<cmath>//ceil(),abs()
#include<algorithm>
#include<iostream>
using namespace std;
char m;
int a[75000], n, x[2] = { 11,21 };
int w = 0, l = 0;
int main()
{
	while (1)
	{
		cin >> m;
		if (m == 'E')
			break;
		if (m == 'W')a[n++] = 1;//1代表华华得1分，0代表对手得1分；
		if (m == 'L')a[n++] = 0;
	}
	for (int j = 0; j < 2; j++)
	{
		w = l = 0;
		for (int i = 0; i < n; i++)
		{
			w += a[i];
			l += 1 - a[i];
			if (max(w, l) >= x[j] && abs(w - l) >= 2)//决出胜负
			{
				cout << w << ":" << l << endl;
				w = l = 0;
			}
		}
		cout << w << ":" << l << endl << endl;
	}
	return 0;
}
```


- 时间复杂度：${O\left( n \right)  }$ 
- 空间复杂度：${O\left( n \right)  }$
