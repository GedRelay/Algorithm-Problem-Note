# 思路
- 思路 1：
使用 `set` 

```cpp
#include<iostream>
#include<set>
using namespace std;
const int inf = 0x7fffffff;
int main()
{
	int n, ans = 0, x;
	cin >> n;
	set <int> s;
	s.insert(inf);
	s.insert(-inf);
	cin >> x;
	s.insert(x);
	ans = x;
	while (--n)
	{
		cin >> x;
		set <int>::iterator it1, it2;
		it1 = s.lower_bound(x);
		it2 = it1;
		it2--;
		if (*it1 == inf)
			ans += abs(*it2 - x);
		else if (*it2 == -inf)
			ans += abs(*it1 - x);
		else
			ans += min(abs(*it2 - x), abs(*it1 - x));
		s.insert(x);
	}
	cout << ans;
	return 0;
}
```
