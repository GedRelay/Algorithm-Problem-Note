# 思路
- 思路 1：
差分


```cpp
#include<iostream>
#include<climits>
using namespace std;

int n, p;
int a[5000010];

int main(){
    scanf("%d%d", &n, &p);
    for(int i = 1; i <= n; i++){
        int x;
        scanf("%d", &x);
        a[i] += x;
        a[i + 1] -= x;
    }
    while(p--){
        int x, y, z;
        scanf("%d%d%d", &x, &y, &z);
        a[x] += z;
        a[y + 1] -= z;
    }
    int ans = INT_MAX;
    for(int i = 1; i <= n; i++){
        a[i] += a[i - 1];
        if(a[i] < ans) ans = a[i];
    }
    cout << ans;
    return 0;
}
```


- 时间复杂度：${O\left( n+p \right)  }$ 
- 空间复杂度：${O\left( n \right)  }$
