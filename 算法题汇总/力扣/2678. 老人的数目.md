```cpp
class Solution {
public:
    int countSeniors(vector<string>& details) {
        int ans = 0;
        for(auto &s : details){
            int age = 0;
            age += 10 * (s[11] - '0');
            age += s[12] - '0';
            if(age > 60) ans ++;
        }
        return ans;
    }
};
```


- 时间复杂度：${O\left( n \right)  }$ 
- 空间复杂度：${O\left( 1 \right)  }$
