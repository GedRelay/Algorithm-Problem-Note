# 思路
- 思路 1：


```cpp
#include<iostream>
#include<cstring>
using namespace std;
struct node
{
	char l, r;
}t[200];

void pre_order(char root)
{
	cout << root;
	if (t[root].l != '*')pre_order(t[root].l);
	if (t[root].r != '*')pre_order(t[root].r);
}

int main()
{
	int n;
	cin >> n;
	char o, root;
	cin >> root;
	cin >> t[root].l >> t[root].r;
	while (--n)
	{
		cin >> o;
		cin >> t[o].l >> t[o].r;
	}//建树
	pre_order(root);//前序遍历
	return 0;
}
```


- 时间复杂度：${O\left( n \right)  }$ 
- 空间复杂度：${O\left( n \right)  }$
