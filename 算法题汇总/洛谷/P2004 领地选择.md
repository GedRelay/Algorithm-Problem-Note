# 思路
- 思路 1：
枚举每个左上角点，利用前缀和快速对区域进行求和
枚举时记录答案


```cpp
#include<iostream>
#include<climits>
#include<cmath>
using namespace std;

int n, m, c;
int mp[1010][1010];
int s[1010][1010];
int main()
{
    cin >> n >> m >> c;
    for (int i = 1; i <= n; i++)
        for (int j = 1; j <= m; j++) {
            cin >> mp[i][j];
            s[i][j] = s[i - 1][j] + s[i][j - 1] - s[i - 1][j - 1] + mp[i][j];
        }
    int maxsum = -1e9, maxi, maxj;
    for (int i = 1; i + c - 1 <= n; i++)
        for (int j = 1; j + c - 1 <= m; j++) {
            int x = i + c - 1, y = j + c - 1;
            int sum = s[x][y] - s[x][j - 1] - s[i - 1][y] + s[i - 1][j - 1];
            if (sum > maxsum) {
                maxsum = sum;
                maxi = i, maxj = j;
            }
        }
    cout << maxi << " " << maxj;
    return 0;
}

```


- 时间复杂度：${O\left( nm  \right)  }$ 
- 空间复杂度：${O\left( nm \right)  }$
