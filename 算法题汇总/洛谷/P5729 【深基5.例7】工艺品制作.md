# 思路
- 思路 1：
使用三维数组模拟


```cpp
#include<iostream>
using namespace std;

int w, x, h, q;
int blocks[21][21][21] = {0}; // 0表示有，1表示无

int main(){
    cin >> w >> x >> h >>q;
    while(q--){
        int x1, y1, z1, x2, y2, z2;
        cin >> x1 >> y1 >> z1 >> x2 >> y2 >> z2;
        for(int i = x1; i <= x2; i++){
            for(int j = y1; j <= y2; j++){
                for(int k = z1; k <= z2; k++){
                    blocks[i][j][k] = 1;
                }
            }
        }
    }
    // 统计答案
    int ans = 0;
    for(int i = 1; i <= w; i++){
        for(int j = 1; j <= x; j++){
            for(int k = 1; k <= h; k++){
                if(blocks[i][j][k] == 0){
                    ans++;
                }
            }
        }
    }
    cout << ans;
    return 0;
}
```


- 时间复杂度：${O\left( wxhq \right)  }$ 
- 空间复杂度：${O\left( wxh \right)  }$
