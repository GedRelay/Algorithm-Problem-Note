# 思路
- 思路 1：


```cpp
#include<iostream>
using namespace std;
int main() 
{
	int a = 0, b = 0, c = 0, pos = 0;
	string s;
	cin >> s;
	while (int(s.find(';', pos)) != -1) {
		pos = s.find(';', pos) + 1;
		char x = s[pos - 5], y = s[pos - 2];
		if ('0' <= y && y <= '9') {
			if (x == 'a')a = y - '0';
			else if (x == 'b')b = y - '0';
			else c = y - '0';
		}
		else {
			if (x == 'a' && y == 'b')a = b;
			else if (x == 'a' && y == 'c')a = c;
			else if (x == 'b' && y == 'a')b = a;
			else if (x == 'b' && y == 'c')b = c;
			else if (x == 'c' && y == 'a')c = a;
			else if (x == 'c' && y == 'b')c = b;
		}
	}
	cout << a << " " << b << " " << c;
	return 0;
}
```


- 时间复杂度：${O\left( 1 \right)  }$ 
- 空间复杂度：${O\left( 1 \right)  }$
