```cpp
#include<iostream>
using namespace std;
struct people
{
	int face;
	string name;
}a[100010];
int main()
{
	int n, m, direct, dist, ops = 0;
	cin >> n >> m;
	for (int i = 0; i < n; i++)
		cin >> a[i].face >> a[i].name;
	for (int i = 0; i < m; i++)
	{
		cin >> direct >> dist;
		if (direct == a[ops].face)//向上数
		{
			ops = ops - dist;
			while (ops < 0)
				ops += n;
		}
		else
			ops = (ops + dist) % n;
	}
	cout << a[ops].name;
	return 0;
}
```

- 空间复杂度：${O\left( n \right)  }$
